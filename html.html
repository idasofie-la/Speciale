Helper functions

# PERT distribution for disability weight uncertainty
rpert <- function(n, min, mode, max) {
  alpha <- 1 + 4 * (mode - min) / (max - min)
  beta  <- 1 + 4 * (max - mode) / (max - min)
  min + rbeta(n, alpha, beta) * (max - min)
}

Monte Carlo

run_mc_row_daly_korea_simple <- function(
  i_diabetes, median_dk, p_detectable_dk,
  D_years, deaths, seyll,
  DW_min, DW_mode, DW_max,
  beta_hat, beta_se, x_ref,
  baseline_uncertainty = FALSE
) {


  # uncertainty in slope
  # draw one possible value of the dose-response slope from a normal distribution centered at the estimated slope, slightly different dose-repsonse slope every timr
  beta <- rnorm(1, mean = beta_hat, sd = beta_se)

  # RR at average detectable Danish exposure, calculate the RR for a person whose exposure is the average detectable Danish exposure
  # Danish detectable average exposure - Korean reference dose
  # log(RR) = beta * (dose - reference dose)
  # should maybe rename to rr_median
  rr_mean <- exp(beta * (median_dk - x_ref))

  # PAF for only detectable population
  PAF <- (p_detectable_dk * (rr_mean - 1)) / (1 + p_detectable_dk * (rr_mean - 1))

  base_cases <- if (baseline_uncertainty) {
    rpois(1, lambda = i_diabetes)
  } else {
    i_diabetes
  }

  # Attributable cases
  AC <- PAF * base_cases

  # draw a disability weight (is also uncertain, therefore PERT)
  DW <- rpert(1, DW_min, DW_mode, DW_max)

  # years lived with disability
  YLD <- AC * D_years * DW
  # attributable deaths
  AD  <- PAF * deaths
  # years of life lost
  YLL <- AD * seyll
  
  DALY <- YLD + YLL

  # return results
  tibble(
    PAF = PAF,
    AC = AC,
    AD = AD,
    YLD = YLD,
    YLL = YLL,
    DALY = DALY
  )
}
run_mc_summary_daly_korea_simple <- function(
  i_diabetes, median_dk, p_detectable_dk,
  D_years, deaths, seyll,
  DW_min, DW_mode, DW_max,
  beta_hat, beta_se, x_ref,
  nsim_outer = 3000,
  baseline_uncertainty = FALSE
) {

  sims <- map_dfr(seq_len(nsim_outer), ~
    run_mc_row_daly_korea_simple(
      i_diabetes = i_diabetes,
      median_dk = median_dk,
      p_detectable_dk = p_detectable_dk,
      D_years = D_years,
      deaths = deaths,
      seyll = seyll,
      DW_min = DW_min,
      DW_mode = DW_mode,
      DW_max = DW_max,
      beta_hat = beta_hat,
      beta_se = beta_se,
      x_ref = x_ref,
      baseline_uncertainty = baseline_uncertainty
    )
  )

  tibble(
    PAF_med  = median(sims$PAF, na.rm = TRUE),
    PAF_lo   = quantile(sims$PAF, 0.025, na.rm = TRUE),
    PAF_hi   = quantile(sims$PAF, 0.975, na.rm = TRUE),

    AC_med   = median(sims$AC, na.rm = TRUE),
    AC_lo    = quantile(sims$AC, 0.025, na.rm = TRUE),
    AC_hi    = quantile(sims$AC, 0.975, na.rm = TRUE),

    AD_med   = median(sims$AD, na.rm = TRUE),
    AD_lo    = quantile(sims$AD, 0.025, na.rm = TRUE),
    AD_hi    = quantile(sims$AD, 0.975, na.rm = TRUE),

    YLD_med  = median(sims$YLD, na.rm = TRUE),
    YLD_lo   = quantile(sims$YLD, 0.025, na.rm = TRUE),
    YLD_hi   = quantile(sims$YLD, 0.975, na.rm = TRUE),

    YLL_med  = median(sims$YLL, na.rm = TRUE),
    YLL_lo   = quantile(sims$YLL, 0.025, na.rm = TRUE),
    YLL_hi   = quantile(sims$YLL, 0.975, na.rm = TRUE),

    DALY_med = median(sims$DALY, na.rm = TRUE),
    DALY_lo  = quantile(sims$DALY, 0.025, na.rm = TRUE),
    DALY_hi  = quantile(sims$DALY, 0.975, na.rm = TRUE)
  )
}
